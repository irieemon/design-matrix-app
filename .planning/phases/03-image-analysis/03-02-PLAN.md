interface NormalizedAnalysis {
  subject: string
  description: string
  textContent: string
  insights: string[]
  relevanceScore: number
}

function normalizeAnalysisResult(raw: any): NormalizedAnalysis {
  return {
    subject: raw.subject || (raw.description?.split('\n')[0]) || 'Image Analysis',
    description: raw.description || raw.projectRelevance || '',
    textContent: raw.textContent || raw.extractedText || '',
    insights: Array.isArray(raw.insights) ? raw.insights : [],
    relevanceScore:
      typeof raw.relevanceScore === 'number' ? raw.relevanceScore
      : raw.relevance === 'high' ? 85
      : raw.relevance === 'medium' ? 55
      : 25,
  }
}

function mapRelevanceToPriority(score: number): IdeaCard['priority'] {
  if (score >= 70) return 'strategic'
  if (score >= 40) return 'moderate'
  return 'low'
}

const ACCEPTED_IMAGE_TYPES = ['image/jpeg', 'image/png', 'image/webp', 'image/gif']
const MAX_PRE_RESIZE_BYTES = 20 * 1024 * 1024

`handleAnalyzeImage()` — the core pipeline:
```typescript
if (!selectedFile || !currentProject?.id) return
setImageError(null)
setImageStage('resizing')
try {
  const resized = await resizeImageToFile(selectedFile, { maxEdgePx: 2048, quality: 0.85 })

  setImageStage('uploading')
  setUploadProgress(10)
  const uploadResult = await FileService.uploadFile(
    resized,
    currentProject.id,
    'Image uploaded for AI analysis',
    currentUser?.id
  )
  if (!uploadResult.success || !uploadResult.file) {
    throw new Error(uploadResult.error || 'Image upload failed. Check your connection and try again.')
  }
  setUploadProgress(70)

  const { data: signedData, error: signedError } = await supabase.storage
    .from('project-files')
    .createSignedUrl(uploadResult.file.storage_path, 3600)
  if (signedError || !signedData?.signedUrl) {
    throw new Error('Could not generate image URL for analysis')
  }
  setUploadProgress(100)

  setImageStage('analyzing')
  const accessToken = localStorage.getItem('sb-access-token') || ''
  const response = await fetch('/api/ai?action=analyze-image', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${accessToken}`,
      ...getCsrfHeaders(),
    },
    body: JSON.stringify({
      imageUrl: signedData.signedUrl,
      projectContext: {
        projectName: currentProject?.name,
        projectDescription: currentProject?.description,
        projectType: currentProject?.project_type,
      },
      analysisType: 'general',
    }),
  })
  if (!response.ok) {
    throw new Error('AI analysis could not be completed. Try uploading a different image or try again later.')
  }
  const json = await response.json()
  const normalized = normalizeAnalysisResult(json.analysis || json)
  setAnalysis(normalized)
  setImageStage('done')
} catch (err) {
  setImageStage('error')
  setImageError(err instanceof Error ? err.message : 'Something went wrong.')
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → Supabase Storage | Direct upload via authenticated Supabase client (RLS + bucket policy) |
| browser → /api/ai | Authenticated + CSRF-protected serverless endpoint |
| /api/ai backend → Supabase signed URL | Server fetches private-bucket image via 1h signed URL |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-10 | Spoofing | POST /api/ai?action=analyze-image | mitigate | Authorization: Bearer header from localStorage; withAuth middleware verifies token server-side (already present, Phase 2) |
| T-03-11 | Tampering (CSRF) | POST /api/ai call from AIIdeaModal | mitigate | `getCsrfHeaders()` from `useCsrfToken()` attaches `X-CSRF-Token`; button disabled while `!hasToken` |
| T-03-12 | Input Validation — malicious file | handleFileSelected | mitigate | MIME whitelist `['image/jpeg','image/png','image/webp','image/gif']` + 20MB size cap BEFORE resize; Supabase bucket `allowedMimeTypes` is second line of defense |
| T-03-13 | Denial of Service (memory) | resizeImageToFile on 25MB file | mitigate | 20MB pre-resize cap rejects oversized files before Canvas allocation |
| T-03-14 | SSRF via imageUrl param | /api/ai backend fetches arbitrary URL | accept | Backend already exists in Phase 2; frontend passes only Supabase-generated signed URLs. Backend-side URL allowlisting is a Phase 7 hardening task, not in this phase's scope. Risk accepted: authenticated users only + CSRF-gated + non-secret private bucket domain. |
| T-03-15 | Information Disclosure — signed URL leakage | createSignedUrl(path, 3600) | mitigate | 1-hour TTL; URL travels only over HTTPS; not logged |

ASVS L1 coverage: V2 (Auth header), V4 (CSRF), V5 (MIME + size validation). No high-severity threats unmitigated.
