       ```typescript
       import type { VercelResponse } from '@vercel/node';
       import {
         withUserRateLimit, withCSRF, withAuth, compose,
         type AuthenticatedRequest,
       } from './_lib/middleware/index.js';
       import {
         handleGenerateIdeas,
         handleGenerateInsights,
         handleGenerateRoadmap,
         handleAnalyzeFile,
         handleAnalyzeImage,
         handleTranscribeAudio,
       } from './_lib/ai/index';

       async function aiRouter(req: AuthenticatedRequest, res: VercelResponse) {
         if (req.method !== 'POST') {
           return res.status(405).json({ error: 'Method not allowed' });
         }

         const action = (req.query.action as string) || '';

         switch (action) {
           case 'generate-ideas': return handleGenerateIdeas(req, res);
           case 'generate-insights': return handleGenerateInsights(req, res);
           case 'generate-roadmap':
           case 'generate-roadmap-v2': return handleGenerateRoadmap(req, res);
           case 'analyze-file': return handleAnalyzeFile(req, res);
           case 'analyze-image': return handleAnalyzeImage(req, res);
           case 'transcribe-audio': return handleTranscribeAudio(req, res);
           default:
             return res.status(404).json({
               error: 'Invalid action',
               validActions: [
                 'generate-ideas', 'generate-insights', 'generate-roadmap',
                 'analyze-file', 'analyze-image', 'transcribe-audio'
               ],
             });
         }
       }

       export default compose(
         withUserRateLimit({ windowMs: 60 * 60 * 1000, maxRequests: 20 }),
         withCSRF(),
         withAuth
       )(aiRouter);
       ```
       - The middleware chain is IDENTICAL to the current one (per D-03)
       - The action dispatch matches all current URL patterns including 'generate-roadmap-v2'
       - The rate limit config matches current values

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client -> api/ai.ts router | All AI requests enter through single router with middleware chain |
| Handler -> AI Gateway | Outbound AI requests carrying prompts with embedded user input |
| transcribeAudio -> audio buffer | Binary audio data from client parsed and forwarded to Whisper |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-09 | Tampering | api/ai.ts thin router | mitigate | Router validates action param against allowlist (switch/case with default 404). Unknown actions rejected. Middleware chain unchanged per D-03. |
| T-02-10 | Information Disclosure | analyzeImage.ts, analyzeFile.ts | mitigate | Image URLs and file content passed to AI API -- ensure no API keys or internal paths leak in prompts. Error responses use existing pattern (debug info only in development). |
| T-02-11 | Tampering | transcribeAudio.ts | mitigate | Audio buffer validation must be preserved from original handler. Do not accept arbitrary file types -- validate content type before passing to Whisper. |
| T-02-12 | Denial of Service | transcribeAudio.ts | mitigate | Two AI API calls per request (transcribe + summarize) doubles cost potential. Rate limiting via withUserRateLimit applies to the router level (single rate limit covers both calls). checkLimit() in handler provides additional per-feature throttling. |
| T-02-13 | Information Disclosure | api/ai.ts error responses | mitigate | Thin router default case returns generic 404 with valid action list only -- no internal state, stack traces, or handler details exposed. |
